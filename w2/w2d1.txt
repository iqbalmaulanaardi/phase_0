Tugas nomor 2

READ and SAVE "year"
IF ("year" modulo 4 == 0)
	DO ELSEIF ("year" modulo 100 == 1)
		DO DISPLAY (Tahun "year" adalah tahun kabisat)
	DO ELSEIF ("year" modulo 100 == 0 AND "year" modulo 400 == 0)
		DO DISPLAY (Tahun "year" adalah tahun kabisat)
	ELSE
		DO DISPLAY (Tahun "year" bukan tahun kabisat)
ELSE
	DO DISPLAY (Tahun "year" bukan tahun kabisat)
